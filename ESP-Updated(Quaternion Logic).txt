// --- NEW HELPER: QUATERNION-BASED ANGLE ---
float quaternionToPitchAngle(float q0, float q1, float q2, float q3) {
  // q0=qw, q1=qx, q2=qy, q3=qz
  
  // This calculates the Pitch angle (rotation around the sensor's Y-axis)
  // Pitch is generally the best single angle for sagittal plane movements like curls.
  
  float sinp = 2.0f * (q0 * q2 - q3 * q1); 
  
  // Use a clamp to avoid domain error near 90 degrees
  if (sinp > 1.0f) sinp = 1.0f;
  if (sinp < -1.0f) sinp = -1.0f;
  
  return asin(sinp) * 180.0f / PI; 
}


// ---------------- IMU SAMPLER ----------------
unsigned long imu_last_ms = 0;
void imuSampleOnce(){
  unsigned long now = millis();
  if(now - imu_last_ms < IMU_INTERVAL_MS) return;
  imu_last_ms = now;

  yield(); // keep watchdog happy

  sensors_event_t a,g,m,t;
  icm.getEvent(&a, &g, &m, &t);

  // Update Madgwick filter (combines gyro/accel for robust orientation)
  filter.updateIMU(g.gyro.x, g.gyro.y, g.gyro.z,
                   a.acceleration.x, a.acceleration.y, a.acceleration.z);

  // raw values
  latest_ax = a.acceleration.x;
  latest_ay = a.acceleration.y;
  latest_az = a.acceleration.z;

  latest_gx = g.gyro.x;
  latest_gy = g.gyro.y;
  latest_gz = g.gyro.z;

  // magnitudes
  float accel_g = sqrt(latest_ax*latest_ax + latest_ay*latest_ay + latest_az*latest_az) / 9.80665f;
  float gyro_mag = sqrt(latest_gx*latest_gx + latest_gy*latest_gy + latest_gz*latest_gz);

  latest_accel_g = accel_g;
  latest_gyro_mag = gyro_mag;

  // Quaternions updated by filter.updateIMU()
  latest_qw = filter.q0; latest_qx = filter.q1; latest_qy = filter.q2; latest_qz = filter.q3;

  // compute quaternion-based tilt angle (Pitch is used for the angle_deg column)
  latest_angle_deg = quaternionToPitchAngle(latest_qw, latest_qx, latest_qy, latest_qz); 
  
  latest_ms = now;
}

// ---------------- HTTP HANDLERS ----------------

  server.send_P(200, "text/html", webpage);
}

void handleData(){
  // snapshot copy
  float ax = latest_ax; float ay = latest_ay; float az = latest_az;
  float gx = latest_gx; float gy = latest_gy; float gz = latest_gz;
  float accel_g = latest_accel_g;
  float gyro_mag = latest_gyro_mag;
  float ang = latest_angle_deg;
  unsigned long ms = latest_ms;

  char buf[512];
  int n = snprintf(buf, sizeof(buf),
    "{\"ms\":%lu,\"angle\":%.6f,"
    "\"accel\":{\"x\":%.6f,\"y\":%.6f,\"z\":%.6f,\"mag\":%.6f},"
    "\"gyro\":{\"x\":%.6f,\"y\":%.6f,\"z\":%.6f,\"mag\":%.6f}}",
    ms, ang,
    ax, ay, az, accel_g,
    gx, gy, gz, gyro_mag);
  if(n > 0) server.send(200, "application/json", buf);
  else server.send(500, "text/plain", "JSON error");
}

// ---------------- WEB MODE ----------------
void runWebServerMode(){
  Serial.println(F(">> WEB MODE"));
  WiFi.begin(ssid, password);
  Serial.print(F("Connecting WiFi"));
  unsigned long start = millis();
  while(WiFi.status() != WL_CONNECTED){
    delay(200);
    Serial.print('.');
    imuSampleOnce();
    yield();
    if(millis() - start > 15000) break;
  }
  Serial.println();
  Serial.print(F("IP: ")); Serial.println(WiFi.localIP());

  server.on("/", handleRoot);
  server.on("/data", handleData);
  server.begin();

  while(digitalRead(SWITCH_PIN) == HIGH){
    imuSampleOnce();
    server.handleClient();
    yield();
    delay(1);
  }

  Serial.println(F("Switch flipped — restarting"));
  ESP.restart();
}

// ---------------- SD MODE ----------------
void runSDMode(){
  Serial.println(F(">> SD MODE"));
  if(!SD.begin(SD_CS)){
    Serial.println(F("SD init failed. Check D8 pin."));
    while(1) delay(1000);
  }
  Serial.println(F("SD OK"));

  // attempt NTP for folder name
  String folder = "";
  bool haveTime = false;
  WiFi.begin(ssid, password);
  int tries = 0;
  while(WiFi.status() != WL_CONNECTED && tries < 20){
    delay(250);
    imuSampleOnce();
    yield();
    Serial.print('.');
    tries++;
  }
  Serial.println();

  if(WiFi.status() == WL_CONNECTED){
    configTime(19800, 0, "pool.ntp.org", "time.nist.gov");
    time_t now = time(nullptr);
    int ttries = 0;
    while(now < 1000 && ttries < 15){
      delay(200);
      imuSampleOnce();
      yield();
      now = time(nullptr);
      ttries++;
    }
    if(now > 1000){
      struct tm timeinfo; localtime_r(&now, &timeinfo);
      char buf[40]; strftime(buf, sizeof(buf), "/%Y-%m-%d_%H-%M-%S", &timeinfo);
      folder = String(buf);
      haveTime = true;
    }
    WiFi.disconnect(true);
  }

  if(!haveTime){
    EEPROM.begin(16);
    int s = EEPROM.read(0);
    s = s + 1;
    if(s < 1 || s > 250) s = 1;
    EEPROM.write(0, s);
    EEPROM.commit();
    sessionNum = s;
    folder = "/session_" + String(s);
  } else {
    EEPROM.begin(16);
    sessionNum = EEPROM.read(0);
  }

  if(!SD.exists(folder)) SD.mkdir(folder);
  String filename = folder + "/imu_log.csv";

  File f = SD.open(filename, FILE_WRITE);
  if(f){
    if(f.size() == 0){
      f.println("ms,angle_deg,ax_mps2,ay_mps2,az_mps2,accel_mag_g,gx_rads,gy_rads,gz_rads,gyro_mag_rads");
    }
    f.close();
  } else {
    Serial.println(F("Failed to open imu log file"));
  }

  Serial.println("Logging to: " + filename);

  unsigned long lastLogMs = 0;

  while(digitalRead(SWITCH_PIN) == LOW){
    imuSampleOnce();
    unsigned long now = millis();
    if(now - lastLogMs >= LOG_INTERVAL_MS){
      lastLogMs = now;

      float ax = latest_ax, ay = latest_ay, az = latest_az;
      float gx = latest_gx, gy = latest_gy, gz = latest_gz;
      float accel_g = latest_accel_g;
      float gyro_mag = latest_gyro_mag;
      float angle = latest_angle_deg;
      unsigned long msstamp = latest_ms;

      File df = SD.open(filename, FILE_WRITE);
      if(df){
        df.printf("%lu,%.6f,%.6f,%.6f,%.6f,%.6f,%.6f,%.6f,%.6f,%.6f\n",
                  msstamp,
                  angle,
                  ax, ay, az,
                  accel_g,
                  gx, gy, gz,
                  gyro_mag);
        df.close();
      } else {
        Serial.println(F("Failed to open log file"));
      }
    }
    yield();
    delay(1);
  }

  Serial.println(F("Switch flipped — restarting"));
  ESP.restart();
}

// ---------------- SETUP ----------------
void setup(){
  Serial.begin(115200);
  delay(200);

  pinMode(SWITCH_PIN, INPUT_PULLUP);
  Wire.begin(SDA_PIN, SCL_PIN);

  if(!icm.begin_I2C(0x68)){
    Serial.println(F("ICM-20948 not found! Check wiring (SDA->D2, SCL->D1)."));
    while(1) delay(10);
  }
  Serial.println(F("ICM-20948 found"));

  icm.setAccelRange(ICM20948_ACCEL_RANGE_4_G);
  icm.setGyroRange(ICM20948_GYRO_RANGE_500_DPS);

  filter.begin(100.0f);

  // Auto-calibrate step (optional) — gather quaternions
  Serial.println(F("Auto-calibrating: hold forearm straight down for 2.5s..."));
  unsigned long t0 = millis();
  unsigned long tEnd = t0 + 2500;
  float qw_acc = 0, qx_acc = 0, qy_acc = 0, qz_acc = 0;
  int cnt = 0;
  while(millis() < tEnd){
    imuSampleOnce();
    qw_acc += filter.q0; qx_acc += filter.q1; qy_acc += filter.q2; qz_acc += filter.q3;
    cnt++;
    delay(10);
    yield();
  }
  if(cnt > 0) Serial.println(F("Calibration stored."));
  else Serial.println(F("Calibration skipped (no data)"));

  // register routes
  server.on("/", handleRoot);
  server.on("/data", handleData);

  // session number from EEPROM
  EEPROM.begin(16);
  sessionNum = EEPROM.read(0);

  // choose mode
  bool webMode = (digitalRead(SWITCH_PIN) == HIGH);
  if(webMode) runWebServerMode();
  else runSDMode();
}
